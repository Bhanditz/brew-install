# Uninstall last version of this and pretend we don't have Clojure
brew uninstall clojure
rm ~/.m2/repository/org/clojure/clojure/1.8.0

# Install (later would be just "brew install clojure")
brew install http://cdn.cognitect.com/brew/clojure.rb

# run a repl
clj

# Create a deps.edn file and start a repl
echo "{:deps {org.clojure/clojure {:type :mvn :version \"1.9.0-alpha17\"}}}" > deps.edn
clj

# Start again - using cached classpath now (.cpcache/default/default.cp), should be faster
clj
